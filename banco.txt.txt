-- Criação da tabela de pedidos dados gerais
CREATE TABLE PedidosDadosGerais (
    NumeroPedido INT AUTO_INCREMENT PRIMARY KEY, -- Número sequencial crescente do pedido
    DataEmissao DATE, -- Data de emissão do pedido
    CodigoCliente INT, -- Chave estrangeira para associar o cliente
    ValorTotal DECIMAL(10, 2) -- Valor total do pedido
);

-- Criação da tabela de pedidos produtos
CREATE TABLE PedidosProdutos (
    Autoincrem INT AUTO_INCREMENT PRIMARY KEY, -- Chave primária única para os produtos do pedido
    NumeroPedido INT, -- Número do pedido associado (chave estrangeira)
    CodigoProduto INT, -- Chave estrangeira para associar o produto
    Quantidade INT, -- Quantidade do produto no pedido
    ValorUnitario DECIMAL(10, 2), -- Valor unitário do produto no pedido
    ValorTotal DECIMAL(10, 2), -- Valor total deste produto no pedido
    FOREIGN KEY (NumeroPedido) REFERENCES PedidosDadosGerais(NumeroPedido), -- FK para associar o pedido
    FOREIGN KEY (CodigoProduto) REFERENCES Produtos(Codigo) -- FK para associar o produto
);

-- Criação de índices nas tabelas (opcional, depende das necessidades específicas)
CREATE INDEX idx_NumeroPedido_PedidosDadosGerais ON PedidosDadosGerais(NumeroPedido);
CREATE INDEX idx_NumeroPedido_PedidosProdutos ON PedidosProdutos(NumeroPedido);
CREATE INDEX idx_CodigoProduto_PedidosProdutos ON PedidosProdutos(CodigoProduto);


-- Criação da tabela de clientes
CREATE TABLE Clientes (
    Codigo INT AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(255) NOT NULL,
    Cidade VARCHAR(100),
    UF CHAR(2)
);

-- Criação da tabela de produtos
CREATE TABLE Produtos (
    Codigo INT AUTO_INCREMENT PRIMARY KEY,
    Descricao VARCHAR(255) NOT NULL,
    PrecoVenda DECIMAL(10, 2) NOT NULL
);
